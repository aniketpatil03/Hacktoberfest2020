	void position(node *p,int key)
	{
		if(!hash[key])
		{
		hash[key] = p;
		}
		else
		{
		node *q;
		q=hash[key];
		while(q->next!='\0')
		q=q->next;
		q->next = p;
		}
	}


	void add()
	{
	node *p;
	p = create();
	position(p,p->key);
		while(p!=NULL)
		{
		cout<<p->name<<":"<<p->phnum<<" "<<"\n";
		p=p->next;
		}

	}


	node* search(int key,string name)
	{
	node *p;
	p=hash[key];
	while((p!='\0')&&(name!=p->name))
	    p=p->next;
	return p;
	}
